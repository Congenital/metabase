import { Canvas, Story, Meta } from "@storybook/addon-docs";
import { Select, Stack } from "metabase/ui";

export const dataWithGroups = [
  { value: 1, label: "Entity key", group: "Overall row" },
  { value: 2, label: "Foreign key", group: "Overall row", disabled: true },
  { value: 3, label: "Category", group: "Common" },
  { value: 4, label: "Description", group: "Common" },
];

export const dataWithIcons = dataWithGroups.map(item => ({
  ...item,
  group: undefined,
}));

export const dataWithLabels = dataWithIcons.map(item => ({
  ...item,
  icon: undefined,
}));

export const args = {
  data: dataWithLabels,
  size: "md",
  label: "Type",
  placeholder: "No semantic type",
  withAsterisk: false,
};

export const sampleArgs = {
  value: 3,
  description: "Determines how Metabase displays the field",
};

export const argTypes = {
  data: {
    control: { type: "json" },
  },
  size: {
    options: ["xs", "md"],
    control: { type: "inline-radio" },
  },
  placeholder: {
    control: { type: "text" },
  },
  withAsterisk: {
    control: { type: "boolean" },
  },
};

<Meta
  title="Inputs/Select"
  component={Select}
  args={args}
  argTypes={argTypes}
/>

# Select

Our themed wrapper around [Mantine Select](https://mantine.dev/core/select/).

## Examples

export const DefaultTemplate = args => <Select {...args} />;

export const VariantTemplate = args => (
  <Stack>
    <Select {...args} />
    <Select {...args} variant="unstyled" />
  </Stack>
);

export const Default = DefaultTemplate.bind({});

<Canvas>
  <Story name="Default">{Default}</Story>
</Canvas>
